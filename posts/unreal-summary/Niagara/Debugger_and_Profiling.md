void UNiagaraDebugDrawModule::DebugDrawBounds(FVectorVMExternalFunctionContext& Context)
{
    VectorVM::FExternalFuncInputHandler<FVector> PositionParam(Context);
    VectorVM::FExternalFuncInputHandler<FVector> SizeParam(Context);

    FDebugDrawInstanceData* InstData = Context.GetInstanceData<FDebugDrawInstanceData>();

    for (int32 i = 0; i < Context.GetNumInstances(); ++i)
    {
        FVector Position = PositionParam.GetAndAdvance();
        FVector Size = SizeParam.GetAndAdvance();

        FBox Bounds(Position - Size * 0.5f, Position + Size * 0.5f);
        InstData->ParticleBounds.Add(Bounds);
    }
}

{

    for (const FBox& Bounds : InstData->ParticleBounds)
    {
        DrawDebugBox(World, Bounds.GetCenter(), Bounds.GetExtent(),

}
